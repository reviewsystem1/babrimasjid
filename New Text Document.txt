<script>
    // script.js content integrated here
    let voteCount = parseInt(localStorage.getItem("totalVotes")) || 50231;
    const scrollArea = document.getElementById("scrollArea");
    const message = document.getElementById("message");
    const countDisplay = document.getElementById("count");

    // Load saved votes for each user
    let userVotes = JSON.parse(localStorage.getItem("userVotes")) || {};

    // 50+ Sample Names
    const sampleNames = [
        "Mohammad Sulaiman","Rizwan","Ayesha","Sharukh","Sameer","Shaik Rahil",
        "Fahad","Imran","Rashid","Zubair","Aadil","Sohail","Tausif","Arshad","Faizan",
        "Afnan","Mubarak","Abdullah","Rehan","Kareem","Arif","Sarfaraz","Naseer",
        "Hameed","Salman","Nawaz","Junaid","Bilal","Aqib","Sajid","Sadiq","Waseem",
        "Shahid","Ikram","Yasir","Armaan","Farhan","Ameen","Aahil","Zaid","Sulaiman",
        "Mannan","Adnan","Ayub","Ismail","Kabir","Hassan","Shuaib","Muzammil","Feroz"
    ];

    // Function to shuffle an array (Fisher-Yates algorithm)
    function shuffleArray(array) {
        for (let i = array.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [array[i], array[j]] = [array[j], array[i]];
        }
    }

    // Create a copy of the names and shuffle it for randomness on refresh
    const shuffledNames = [...sampleNames];
    shuffleArray(shuffledNames);

    // Duplicate the SHUFFLED list for infinite scrolling
    const movingList = [...shuffledNames, ...shuffledNames];

    // Populate the scrolling area
    movingList.forEach(name => {
        let div = document.createElement("div");
        div.textContent = name + " has voted";
        scrollArea.appendChild(div);
    });

    // Show initial total votes
    countDisplay.textContent = "Total Votes: " + voteCount.toLocaleString();

    // LIKE function
    function likeFunction() {
        let name = document.getElementById("username").value.trim();
        if (name === "") return alert("Please enter your name!");

        message.textContent = `${name} has liked ‚ù§Ô∏è`;

        let div = document.createElement("div");
        div.textContent = `${name} has liked`;
        scrollArea.prepend(div);
    }

    // VOTE function
    function voteFunction() {
        let name = document.getElementById("username").value.trim();
        if (name === "") return alert("Please enter your name!");

        if (!userVotes[name]) userVotes[name] = 0;

        // Block more than 3 votes
        if (userVotes[name] >= 3) {
            alert("‚ùó You have already voted 3 times.");
            return;
        }

        // Count this vote
        userVotes[name]++;
        localStorage.setItem("userVotes", JSON.stringify(userVotes));

        // Increase total votes
        voteCount++;
        localStorage.setItem("totalVotes", voteCount);
        
        // Display total votes
        countDisplay.textContent = "Total Votes: " + voteCount.toLocaleString();

        message.textContent = `${name} has voted üó≥`;

        let div = document.createElement("div");
        div.textContent = `${name} has voted`;
        scrollArea.prepend(div);
    }
